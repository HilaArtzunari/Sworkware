
	\begin{UseCase}{CU21}{Denunciar usuario}{
		Permite a un usuario denunciar a otro usuario por comportamiento indevido.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Usuario}
		\UCitem{Propósito}{Registrar el comportamiento fraudolento de los usuarios del sistema.}
		\UCitem{Resumen}{
		El sistema da la opción de denunciar a usuarios que comentan fraudes.}
		\UCitem{Entradas}{
		Click en Denunciar usuario dentro de la ventana del perfil del usuario a denunciar.}
		\UCitem{Salidas}{Una ventana con tres opciones a elegir.}
		\UCitem{Precondiciones}{
		Estar como usuario del sistema, haber entrado en la ventana de ver perfil.}
		\UCitem{Postcondiciones}{Un correo al administrador del sistema.}
		\UCitem{Autor}{Angel Ferreira Osorno.}
		\UCitem{Revisor}{Roberto Mendoza Saavedra.}
	\end{UseCase}
