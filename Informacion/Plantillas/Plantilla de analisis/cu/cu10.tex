
	\begin{UseCase}{CU10}{Editar informacion de perfil.}{
		Permite editar la informacion de un perfil como la escuela de procedencia, nombre de perfil y cambiar la imagen de perfil.
	}
		\UCitem{Versión}{1.0}
		\UCitem{Actor}{Usuario}
		\UCitem{Propósito}{Editar la información y/o imagen del perfil.}
		\UCitem{Entradas}{Imagen,escuela de procedencia, nombre de perfil, hypervínculos}
		\UCitem{Origen}{Teclado,mouse.}
		\UCitem{Salidas}{Ventana con un mesaje de modificación exitosa.}
		\UCitem{Destino}{Pantalla.}
		\UCitem{Precondiciones}{Estar registrado como usuario del sitio,no estar bloqueado.}
		\UCitem{Postcondiciones}{Los datos como escuela de procedencia, nombre de perfil y/o imagen cambian en el perfil.}
		\UCitem{Errores}{{\bf E1:} ``Error de conexión al servidor.'' -- El sistema muestra el Mensaje {\bf MSG1-}``Inténtalo más tarde.'' y continua al paso 3.}
		
		\UCitem{Tipo}{Caso de uso primario}
		\UCitem{Observaciones}{}
		\UCitem{Autor}{Angel Ferreira Osorno.}
		\UCitem{Revisor}{Diana Laura Mejía Mendoza.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] da click con el cursor en el botón \IUbutton{Editar perfil} en la \IUref{IU9}{Perfil de usuario}.
		\UCpaso abre la pantalla para editar los datos del perfil \IUref{IU10}{Editar perfil}.
		\UCpaso[\UCactor] llena los campos requeridos en los cuadros de texto en \IUref{IU10}{Editar perfil}\Trayref{C}.
		\UCpaso Habilita el botón \IUbutton{Guardar} de la \IUref{IU10}{Editar perfil}.
		\UCpaso Guarda los datos en los cuadros de texto de la \IUref{IU10}{Editar} \Trayref{A} \Trayref{B}.
		\UCpaso Muestra un mensaje que la modificaión fue exitosa.
		\UCpaso Actualiza la información de la \IUref{IU9}{Perfil de usuario} con los nuevos valores ingresados.
		
	\end{UCtrayectoria}
		\begin{UCtrayectoriaA}{A}{Falla en la conexión con el servidor.}
			\UCpaso Muestra el Mensaje {\bf MSG1-}``Tú información no puede ser actualizada en este momento. Inténtalo más tarde.''.
			\UCpaso Continúa en el paso 3 del \UCref{CU10}.
		\end{UCtrayectoriaA}

		\begin{UCtrayectoriaA}{B}{No hay contenido en la caja de texto de los datos del perfil.}
			\UCpaso Deshabilita el botón \IUbutton{Guardar} de la \IUref{IU10}{Editar perfil}.
			\UCpaso Continúa en el paso 2 del \UCref{CU10}.
		\end{UCtrayectoriaA}

		
		\begin{UCtrayectoriaA}{C}{Cancelar.}
			\UCpaso[\UCactor]da clic al boton \IUbutton{Cancelar} en la \IUref{IU10}{Editar perfil}
			\UCpaso Continúa en el paso 2 del \UCref{CU9}.
		\end{UCtrayectoriaA}
