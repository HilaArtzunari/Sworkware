	\begin{UseCase}{CU12}{Enviar comentario en la oferta de venta.}{
		Permite enviar un comentario en la oferta de venta de un componente electrónico para poder interactuar con el vendedor. Sólo se podrá enviar si el usuario se encuentra registrado.
	}
		\UCitem{Versión}{1.0}
		\UCitem{Actor}{Comprador y vendedor.}
		\UCitem{Propósito}{Tener una interacción con el vendedor al momento de que un comprador se encuentre interesado en el componente electrónico.}
		\UCitem{Entradas}{Comentario.}
		\UCitem{Origen}{Teclado.}
		\UCitem{Salidas}{Comentario debajo de la oferta de venta}
		\UCitem{Destino}{Pantalla.}
		\UCitem{Precondiciones}{Estar registrado como usuario del sitio.}
		\UCitem{Postcondiciones}{La oferta tendrá un historial de venta.}
		\UCitem{Errores}{{\bf E1:} ``Error de conexión al servidor.'' -- El sistema muestra el Mensaje {\bf MSG1-}``Tú comentario no puede ser enviado en este momento. Inténtalo más tarde.'' y continua al paso 3.}
		
		\UCitem{Tipo}{Caso de uso primario}
		\UCitem{Observaciones}{}
		\UCitem{Autor}{Roberto Mendoza Saavedra.}
		\UCitem{Revisor}{Gabriela Moreno González}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Ingresa el comentario en la caja de texto de la \IUref{IU5}{Oferta de venta}.
		\UCpaso Habilita el botón \IUbutton{Enviar} de la \IUref{IU5}{Oferta de venta}.
		\UCpaso[\UCactor] Solicita el envío del comentario presionando el botón \IUbutton{Enviar} de la \IUref{IU5}{Oferta de venta} \Trayref{A}.
		\UCpaso Registra el comentario enviado por el comprador o vendedor.
		\UCpaso Muestra el comentario debajo de la oferta mediante la \IUref{IU5}{Oferta de venta}
\end{UCtrayectoria}
		\begin{UCtrayectoriaA}{A}{Falla en la conexión con el servidor.}
			\UCpaso Muestra el Mensaje {\bf MSG1-}``Tú comentario no puede ser enviado en este momento. Inténtalo más tarde.''.
			\UCpaso Continúa en el paso 3 del \UCref{CU12}.
		\end{UCtrayectoriaA}
