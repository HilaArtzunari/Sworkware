
	\begin{UseCase}{CU20}{Ver Reputación}{
		Permite al interesado ver la reputación del dueño del componente.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Intereserado}
		\UCitem{Propósito}{Que el interesado del componente obtenga una perspectiva del historial del dueño del componente.}
		\UCitem{Resumen}{
		El sistema mostrara un Rankin stars para mostrar un promedio de las transacciones exitosas del dueño del componente.}
		\UCitem{Entradas}{
		Click en ver perfil dentro de la ventana del componente.
		}
		\UCitem{Salidas}{Un Ranking stars dentro de la información del perfil del dueño del componente.}
		\UCitem{Precondiciones}{
		Estar como usuario del sistema, haber entrado en la ventana de ver componente.}
		\UCitem{Postcondiciones}{ninguna.}
		\UCitem{Autor}{Angel Ferreira Osorno.}
		\UCitem{Revisor}{Roberto Mendoza Saavedra.}
	\end{UseCase}
