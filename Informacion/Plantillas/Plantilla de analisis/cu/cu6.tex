
	\begin{UseCase}{CU6}{Calificar usuario}{
		Permite al interesado en un componente calificar al dueño de dicho componente después de contactarlo.
	}
		\UCitem{Versión}{0.1}
		\UCitem{Actor}{Intereserado}
		\UCitem{Propósito}{Generar una reputación en el dueño del componente.}
		\UCitem{Resumen}{
		El sistema activará calificar un dueño despues que el interesado haya activado la opción de contactar en el sistema.}
		\UCitem{Entradas}{
		- Star Ranking: número entero con valor entre 0 a 5.
		}
		\UCitem{Salidas}{El promedio en el Star Ranking con las calificaciones anteriores.}
		\UCitem{Precondiciones}{
		Estar como usuario del sistema, haber habilitado el botón de contactar en el sistema.
		}
		\UCitem{Postcondiciones}{Se registrará el puntaje en la base de datos.}
		\UCitem{Autor}{Angel Ferreira Osorno.}
		\UCitem{Revisor}{Roberto Mendoza Saavedra.}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Abre el correo recibido con el asunto 'Califica tu experiencia en la compra del componente electrónico name\_component'
		\UCpaso[\UCactor] Oprime el botón \IUbutton{Escribir opinión} \Trayref{A}
		\UCpaso Abre una nueva pestaña con la pantalla \IUref{UI21}{Calificar usuario} 
		\UCpaso[\UCactor] Ingresa su calificación y su comentario en la nueva pestaña.
		\UCpaso[\UCactor] Oprime el botón \IUbutton{Enviar comentario} 
		\UCpaso Guarda tanto la calificación como el comentario.
		\UCpaso Muestra el mensaje {\bf MSG22-}``Gracias por tu opinión.''
	\end{UCtrayectoria}
		
		\begin{UCtrayectoriaA}{A}{El usuario no ha iniciado sesión}
			\UCpaso Muestra la pantalla \IUref{UI1}{Login}
			\UCpaso[\UCactor] Realiza el caso de uso \UCref{CU1}
			\UCpaso[\UCactor] Continúa en el paso 3 del \UCref{CU6}
		\end{UCtrayectoriaA}
