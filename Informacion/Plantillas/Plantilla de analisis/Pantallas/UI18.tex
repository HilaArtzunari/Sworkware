\subsection{UI18 Perfil de administrador}

\subsubsection{Objetivo}
	Mostrar las opciones de administración.

\subsubsection{Diseño}
	Esta pantalla aparece cuando el administrador da clic en el botón \IUbutton{Perfil} y muestra una lista de todas las opciones de administración que posee.

\IUfig[.5]{gui/UI18}{UI18}{Perfil de administrador}

\subsubsection{Salidas}

	Ninguna.

\subsubsection{Entradas}

	Ninguna.

\subsubsection{Comandos}
\begin{itemize}
	\item \IUbutton{Ver usuarios registrados}: Muestra una lista de todos los usuarios registrados.
	\item \IUbutton{Eliminar un usuario}: Elimina a un usuario mediante el correo electrónico.
	\item \IUbutton{Ver denuncias}: Muestra una lista de todas las denuncias realizadas.
	\item \IUbutton{Registrar un nuevo usuario}: Muestra la pantalla para el registro de usuarios.
	 \item \IUbutton{Editar información de una cuenta de usuario}: Muestra la \IUref{IU4}{Perfil de usuario}
	 \item \IUbutton{Registrar un nuevo administrador}: Muestra un formulario para el registro de un nuevo administrador.
	 \item \IUbutton{Bloquear un usuario}: Bloquea a un usuario mediante el correo electrónico.
\end{itemize}

\subsubsection{Mensajes}
	\begin{Citemize}
		\item {\bf MSG100} Error al conectarse con el servidor, por favor intente más tarde.
	\end{Citemize}
