\subsection{IU4 Eliminar componente}

\subsubsection{Objetivo}
	Realiza la eliminación de un registro de componente del sistema.

\subsubsection{Diseño}
	Esta pantalla aparece en la secci\'on de ''Mis componentes''. 

\IUfig[.5]{gui/UI4}{IU23}{Pantalla Eliminiar componente.}

\subsubsection{Salidas}

	Ninguna.

\subsubsection{Entradas}
Selecci\'on del componente o los componentes a eliminar.

\subsubsection{Comandos}
\begin{itemize}
	\item \IUbutton{Eliminar}: Elimina componente
	
	%Verifica que el Estudiante se encuentre registrado y la contraseña sea la correcta. Si la verificación es correcta, se muestra la \IUref{UI32}{Pantalla de Selección de Seminario}.
	\item \IUbutton{Regresar}: Regresa a la pagina principal

	%Muestra la ayuda de esta pantalla \IUref{UI50}{Pantalla de Ayuda}.
\end{itemize}

\subsubsection{Mensajes}
	\begin{Citemize}
		\item {\bf MSG5} Se ha eliminado correctamente el componente.
	\end{Citemize}
